Ejemplos de Tests

Carrito

it("un usuario no ve el carrito de otro", async () => {
  const tokenA = await login("a@test.com","123");
  const tokenB = await login("b@test.com","123");
  await request(app).post("/carritos/detalle")
    .set("Authorization",`Bearer ${tokenA}`)
    .send({ id_producto:1, cantidad:1 });
  const resB = await request(app).get("/carritos/me")
    .set("Authorization",`Bearer ${tokenB}`);
  expect(resB.body.items).toHaveLength(0);
});

Favoritos

it("rechaza favorito duplicado", async () => {
  const token = await login("c@test.com","123");
  await request(app).post("/favoritos")
    .set("Authorization",`Bearer ${token}`)
    .send({ id_producto:2 });
  const res = await request(app).post("/favoritos")
    .set("Authorization",`Bearer ${token}`)
    .send({ id_producto:2 });
  expect(res.statusCode).toBe(409);
});

Rese√±as

it("no acepta comentario >300 chars", async () => {
  const token = await login("d@test.com","123");
  const comentario = "a".repeat(301);
  const res = await request(app).post("/resenas/producto/1")
    .set("Authorization",`Bearer ${token}`)
    .send({ comentario, calificacion:5 });
  expect(res.statusCode).toBe(400);
});
